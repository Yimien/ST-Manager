      function toggleCollapsible() {
        const collapsible = document.getElementById('sakuraCollapsible');
        const header = collapsible.querySelector('.sakura-header');
        const isExpanded = collapsible.classList.toggle('expanded');
        header.setAttribute('aria-expanded', isExpanded ? 'true' : 'false');

        // 展开时触发花瓣飘落特效
        if (isExpanded) {
          triggerPetalFall();

        // 配合 0.5s 的 CSS transiton 动画时长，在动画开始和结束时都更新高度
        setTimeout(updateIframeHeight, 50);
        setTimeout(updateIframeHeight, 550);
      }

      // 生成飘落花瓣的独立函数
      function triggerPetalFall() {
        const contentBox = document.querySelector('.sakura-content-wrapper'); // 放到内部以免溢出问题
        const petalCount = 8;
        const emojis = ['🌸', '💮'];

        for (let i = 0; i < petalCount; i++) {
          const petal = document.createElement('div');
          petal.className = 'petal-particle';
          petal.innerText = emojis[Math.floor(Math.random() * emojis.length)];

          const size = 10 + Math.random() * 8;
          const startX = 10 + Math.random() * 80;
          const duration = 1.5 + Math.random() * 1.5;
          const delay = Math.random() * 0.5;

          Object.assign(petal.style, {
            fontSize: `${size}px`,
            left: `${startX}%`,
            animationDuration: `${duration}s`,
            animationDelay: `${delay}s`
          });

          contentBox.appendChild(petal);

          setTimeout(() => {
            if (petal.parentNode === contentBox) {
              contentBox.removeChild(petal);
            }
          }, (duration + delay) * 1000);
        }
      }

      function updateIframeHeight() {
        const bodyHeight = document.body.scrollHeight;
        if (window.parent && window.parent !== window) {
          window.parent.postMessage({ type: 'resizeIframe', height: bodyHeight }, '*');
        }
      }

      document.addEventListener('keydown', function (event) {
        if (event.key === 'Enter' || event.key === ' ') {
          const activeElement = document.activeElement;
          if (activeElement && activeElement.classList.contains('sakura-header')) {
            event.preventDefault();
            toggleCollapsible();
          }
        }
      });

      window.addEventListener('load', updateIframeHeight);
      window.addEventListener('resize', updateIframeHeight);

一点淡淡的、专属于森林清晨的草木清香从她袖口散发出来，令人安心。翠绿色的光点从她的指尖溢出，顺着我的眉心没入思绪。

木屋里的炉火噼啪作响，火光映照在她白皙的皮肤上，为她镀上了一层暖洋洋的色泽。

“原来是这种美妙的语言呀。现在，你应该能听懂我说话了吧？”

塞拉菲娜的唇边勾起好看的线条，嘴里吐出的已经是字正腔圆且温柔的中文。她收回手，替我将被角往上拉了拉，盖住我失去血色的肩膀。

“重新向你问好，我是塞拉菲娜，这片森林的守护者。”她低下头，凑得离我近了些，那股花草混合着她特有体香的味道更加清晰，“你受了很重的伤，那些暗影獠牙差点就彻底撕裂了你。我用魔法修补了你的身躯，但你还需要很久的修养。”

“来，喝下这杯安神茶。它能驱散你体内的寒气，也能抚平你遭遇袭击后的恐惧。”她耐心地举着杯子，满目柔情地注视着我，“你叫什么名字？又为何会独自闯入这片充满危险的埃尔多里亚呢？”
